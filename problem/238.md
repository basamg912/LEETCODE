```cpp
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int total=1;
        int size = nums.size();
        vector<int> zero_indices;
        vector<int> answer(size);
        bool has_zero = false;
        for (int i=0; i<size; i++){
            int num = nums[i];
            if (num ==0){
                zero_indices.push_back(i);
                has_zero = true;
                continue;
            }
            total *= num;
        }
        if (has_zero){
            if (zero_indices.size() > 1){
                return answer;
            }
            answer[zero_indices[0]] = total;
            return answer;
        }
        for (int i=0; i<size; i++){
            answer[i] = total/nums[i];
        }
        return answer;

    }
};
/*
새로운 vector 를 반환, answer[i] 는 nums[i] 를 제외한 나머지 원소들의 곱을 반환,
O(N) 복잡도로 해결하는게 핵심 구현 포인트 division operation 없이

[1,2,3,4]
1. 2*3*4 = 24
2. 2*3*4 / 2 * 1 = 12
3. (2*3*4 / 2*1) /3 *2
...
*/
```

Approach
- 0 이 두개 이상 있으면 answer 는 다 0 인거니깐, answer 자체를 반환. (0으로 알아서 초기화되어있다.)
- 0 이 한개일때에만 total 계산한거 0 index 에 넣어주고, 나머지 0 으로 채워주기. 
- 0 이 없을때에는 total 을 기준으로 nums[i] 로 하나씩 나눠서 넣어주기

Approach2
- 나눗셈 없이, edge case 처리 없이 구하는 방법
- prefix, suffix 배열을 생성해서 처리

```cpp
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int size = nums.size();
        vector<int> pre(size, 1);
        vector<int> suff(size, 1);
        vector<int> answer(size);

        for (int i=1; i<size; i++){
            pre[i] = pre[i-1] * nums[i-1]; // pre[1] 은 1번 원소 전까지 원소곱
        }
        for (int i=size-2; i>=0; i--){
            suff[i] = suff[i+1]*nums[i+1];
        }
        for (int i=0; i<size; i++){
            answer[i] = pre[i] * suff[i];
        }
        return answer;
    }
};
```
- 왼쪽부터 계산한 누적곱 x 오른쪽부터 계산한 누적곱
