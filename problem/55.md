jump game
list 에 있는 각 원소들이 몇 칸 점프할지를 의미
첫번째 인덱스에서 시작 마지막 원소 도달 가능할지 bool return
- 각 인덱스의 숫자는 가능한 최대 점프 거리를 의미

`[2,5,0,0]` 도 true, 왜냐하면 첫번째칸에서 1번만 뛰어도 됨 ㅇㅇ
첫번째 인덱스부터 시작해서 0 이 만들어지는지 검사하면될듯? -> maxlen 을 이용해서 업데이트

Approaceh 1 - greedy
```cpp
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int reachable_index = 0;
        int size = nums.size(); // 원소 갯수
        int last_index = size-1;
        int place = 0;
        while (reachable_index <= last_index){
            if (nums[place] == 0 && reachable_index < last_index){
                return false;
            }
            // [3,2,1,0,4] -> 처음 range = 3 -> 0,1,2,3 번 인덱스까지 접근가능
            int range = nums[place];
            for (int i=0; i<range; i++){
                int place2 = place+i;
                int range2 = nums[place2];
                if (range2 > last_index)    return true;
                reachable_index = max(reachable_index, place2 + range2);
            }
            place = reachable_index;
        }
        return true;
    }
};
```
- 위 방식의 한계는 단순히, 가장 멀리 도달하는 것만을 목표로 하다보니, 실제 도달 불가능한 경우가 발생 가능

Approach 2
처음에는 앞에서부터 접근 -> 뒤에서부터 ㄱㄱ
```cpp
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int goal = nums.size() -1;
        for (int pos = goal-1; pos >= 0 ; pos--){
            if (pos + nums[pos] >= goal){
                goal = pos;
            }
        }
        return goal == 0;
    }
};
```
